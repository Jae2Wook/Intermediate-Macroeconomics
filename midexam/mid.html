Knowledge[i] = Knowledge[i] * 2
new_k = (1 - l) * (s2 / (d + g + n))**(1 / (1 - alpha))

while abs(k[i] - new_k) > 0.01:
    Capital= np.append(Capital, Capital[i] + (s2 * Output[i] - d * Capital[i]))
    Labor= np.append(Labor, Labor[i] * (1 + n))
    Knowledge= np.append(Knowledge, Knowledge[i] * (1 + g))
    Output= np.append(Output, RS(l, Knowledge[i + 1], Capital[i + 1], Labor[i + 1], alpha))
    Consumption= np.append(Consumption, (1 - s2) * Output[i + 1])
    k= np.append(k, Capital[i + 1] / (Knowledge[i + 1] * Labor[i + 1]))
    i += 1
    time = np.append(time, i)

print("new steady-state: {:.2f}".format(new_k))
